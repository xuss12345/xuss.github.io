const int N=1005,M=20005;
int n,m,tot,fa[N];
struct Edge{int u,v,w;}e[M];

bool cmp(Edge a,Edge b){return a.w<b.w;}
int find(int x){return fa[x]==x?x:fa[x]=find(fa[x]);}

int kruskal(){
    sort(e+1,e+m+1,cmp);
    for(int i=1;i<=n;i++)fa[i]=i;
    int ans=0;
    for(int i=1;i<=m;i++){
        int fu=find(e[i].u),fv=find(e[i].v);
        if(fu!=fv){
            fa[fu]=fv;
            ans+=e[i].w;
            tot++;
            if(tot==n-1)break;
        }
    }
    return ans;
}
