三、基础版：自动消息发送器

我们先从最简单的单次消息发送开始：

import itchat
import time

def send_single_message():
    """
    发送单条消息给指定好友
    """
    # 自动登录，hotReload=True可以缓存登录状态
    itchat.auto_login(hotReload=True)

    # 通过备注名查找好友
    friend = itchat.search_friends(name='张三')
    if friend:
        # 发送文本消息
        itchat.send('您好，这是自动发送的测试消息！', friend[0]['UserName'])
        print(f"消息发送成功给：{friend[0]['NickName']}")
    else:
        print("未找到指定好友")

if __name__ == "__main__":
    send_single_message()

这段代码实现了最基本的消息发送功能。但实际使用时，我们需要更健壮的版本：

import itchat
import time
from typing import List

class WeChatAutoSender:
    def __init__(self):
        self.is_logged_in = False

    def login(self):
        """登录微信"""
        try:
            itchat.auto_login(hotReload=True, enableCmdQR=2)
            self.is_logged_in = True
            print("微信登录成功！")
            return True
        except Exception as e:
            print(f"登录失败：{e}")
            return False

    def send_to_friend(self, friend_remark: str, message: str) -> bool:
        """
        给指定好友发送消息

        Args:
            friend_remark: 好友备注名
            message: 要发送的消息内容

        Returns:
            bool: 发送是否成功
        """
        if not self.is_logged_in:
            if not self.login():
                return False

        try:
            friends = itchat.search_friends(name=friend_remark)
            if not friends:
                print(f"未找到备注为'{friend_remark}'的好友")
                return False

            friend = friends[0]
            itchat.send(message, friend['UserName'])
            print(f"消息发送成功给：{friend['NickName']}({friend_remark})")
            return True
        except Exception as e:
            print(f"发送消息失败：{e}")
            return False

    def send_to_group(self, group_name: str, message: str) -> bool:
        """
        给指定群聊发送消息
        """
        if not self.is_logged_in:
            if not self.login():
                return False

        try:
            groups = itchat.search_chatrooms(name=group_name)
            if not groups:
                print(f"未找到群聊：{group_name}")
                return False

            group = groups[0]
            itchat.send(message, group['UserName'])
            print(f"群消息发送成功：{group_name}")
            return True
        except Exception as e:
            print(f"发送群消息失败：{e}")
            return False

# 使用示例
if __name__ == "__main__":
    sender = WeChatAutoSender()

    # 给好友发送消息
    sender.send_to_friend("张三", "下午3点开会，请不要迟到！")

    # 给群聊发送消息
    sender.send_to_group("家庭群", "今晚我加班，不回家吃饭了")

四、进阶版：智能定时消息系统

单次发送还不够智能，我们来打造一个支持定时任务的完整系统：

import itchat
import schedule
import time
import threading
from datetime import datetime
import json
import os

class AdvancedWeChatAssistant:
    def __init__(self, config_file='task_config.json'):
        self.config_file = config_file
        self.tasks = []
        self.is_running = False
        self.load_config()

    def load_config(self):
        """从配置文件加载任务"""
        if os.path.exists(self.config_file):
            try:
                with open(self.config_file, 'r', encoding='utf-8') as f:
                    self.tasks = json.load(f)
                print(f"加载了 {len(self.tasks)} 个定时任务")
            except Exception as e:
                print(f"加载配置文件失败：{e}")
                self.tasks = []

    def save_config(self):
        """保存任务到配置文件"""
        try:
            with open(self.config_file, 'w', encoding='utf-8') as f:
                json.dump(self.tasks, f, ensure_ascii=False, indent=2)
        except Exception as e:
            print(f"保存配置失败：{e}")

    def add_daily_task(self, task_name: str, target: str, message: str,
                      target_time: str, is_group: bool = False):
        """
        添加每日定时任务

        Args:
            task_name: 任务名称
            target: 目标好友或群聊名称
            message: 消息内容
            target_time: 执行时间，格式：HH:MM
            is_group: 是否为群聊
        """
        task = {
            'name': task_name,
            'target': target,
            'message': message,
            'time': target_time,
            'is_group': is_group,
            'enabled': True
        }

        self.tasks.append(task)
        self.save_config()
        print(f"已添加任务：{task_name}")

    def send_message(self, target: str, message: str, is_group: bool = False):
        """发送消息的核心方法"""
        try:
            if is_group:
                chats = itchat.search_chatrooms(name=target)
            else:
                chats = itchat.search_friends(name=target)

            if not chats:
                print(f"未找到目标：{target}")
                return False

            chat = chats[0]
            itchat.send(message, chat['UserName'])
            print(f"{datetime.now().strftime('%Y-%m-%d %H:%M:%S')} "
                  f"消息发送成功给：{target}")
            return True
        except Exception as e:
            print(f"发送消息失败：{e}")
            return False

    def setup_schedule(self):
        """设置定时任务"""
        for task in self.tasks:
            if not task['enabled']:
                continue

            time_str = task['time']
            target = task['target']
            message = task['message']
            is_group = task['is_group']

            # 使用闭包保存参数
            def create_job(t=target, m=message, g=is_group):
                return lambda: self.send_message(t, m, g)

            schedule.every().day.at(time_str).do(create_job())
            print(f"已安排任务：{task['name']}，每天{time}执行")

    def start(self):
        """启动助手"""
        print("正在登录微信...")
        itchat.auto_login(hotReload=True, enableCmdQR=2)
        print("微信登录成功！")

        self.setup_schedule()
        self.is_running = True

        print("微信助手已启动，开始监控定时任务...")

        # 在单独的线程中运行定时任务检查
        def run_schedule():
            while self.is_running:
                schedule.run_pending()
                time.sleep(1)

        schedule_thread = threading.Thread(target=run_schedule)
        schedule_thread.daemon = True
        schedule_thread.start()

        # 保持主线程运行
        itchat.run()

    def stop(self):
        """停止助手"""
        self.is_running = False
        schedule.clear()
        print("微信助手已停止")

# 使用示例
def main():
    assistant = AdvancedWeChatAssistant()

    # 添加示例任务（实际使用时可以根据需要修改）
    if not assistant.tasks:
        # 工作日早上9点给老板发日报
        assistant.add_daily_task(
            "工作日日报",
            "王总",
            "王总早上好！这是今天的工作计划：1. 完成项目文档 2. 团队会议 3. 代码评审",
            "09:00"
        )

        # 每天晚上8点家庭群报平安
        assistant.add_daily_task(
            "家庭报平安",
            "幸福一家人",
            "大家晚上好，我今天一切都好，你们呢？",
            "20:00",
            is_group=True
        )

    # 启动助手
    try:
        assistant.start()
    except KeyboardInterrupt:
        print("\n收到停止信号，正在退出...")
        assistant.stop()

if __name__ == "__main__":
    main()

class MessageTemplate:
    @staticmethod
    def daily_report_template(name, tasks):
        """日报模板"""
        task_list = "\n".join([f"{i+1}. {task}" for i, task in enumerate(tasks)])
        return f"""{name}的工作日报：

今日完成：
{task_list}

明日计划：
1. 待规划

遇到的问题：无"""

    @staticmethod
    def meeting_reminder(meeting_name, time, location):
        """会议提醒模板"""
        return f"""会议提醒：
主题：{meeting_name}
时间：{time}
地点：{location}
请准时参加！"""

# 使用模板
template = MessageTemplate()
message = template.daily_report_template("张三", ["完成需求分析", "编写代码", "单元测试"])
print(message)
